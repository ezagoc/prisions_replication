\section{Motivation}

\begin{frame}
\frametitle{Motivation}

\small
\begin{itemize}\setlength\itemsep{0.65em}
\item  Prison overcrowding is a global phenomenon, in 2021, 118 countries reported prison populations exceeding official capacity {\footnotesize\color{gray}\citep{GlobalPrisonTrends}}
\pause
\item It is often framed as a \alert{state capacity problem}, leading many policymakers to view the appropriate response as a \textbf{further expansion of state capacity}.
\pause
\item However, actual evidence of this is scarce since: 
\begin{enumerate}\setlength\itemsep{0.65em}
    \item \textbf{criminal justice outcomes depend on many actors: judges, prosecutors, police, prison guards.}
    \pause
    \item \textbf{increases in capacity are endogenous to prior levels and incarceration rates.}
\end{enumerate}
\pause
\item General equilibrium effects are often overlooked.
\end{itemize}
\end{frame}

\section{Research Question}
\begin{frame}
\frametitle{Research Question and Strategy}
\small
\begin{itemize}\setlength\itemsep{0.65em}
    \item Capacity-expanding policies may indirectly shape the behavior of these agents — raising fundamental questions about \textbf{institutional constraints and incentives}.
\end{itemize}

\pause
\vspace{0.5em}
\begin{center}
    \normalsize \textbf{RQ: Do judges sentence more individuals to prison when capacity is increased?}
\end{center}
\vspace{0.5em}

\pause

\begin{itemize}\setlength\itemsep{0.65em}
    \item I exploit the staggered construction of \textbf{federal} prisons across Mexico as plausibly exogenous shocks to \textbf{local} prison capacity trends.
    \pause
    \item The identification strategy relies on three institutional features: 
    \begin{enumerate}\setlength\itemsep{0.65em}
        \item Judges must assign inmates to the nearest available facility.\pause
        \item State prisons also house federal prisoners.\pause
        \item New federal prisons were built to relocate federal inmates — not to address local capacity constraints.
    \end{enumerate}
\end{itemize}
\end{frame}


\section{Contribution}
\begin{frame}
\frametitle{Contribution}
\small

\begin{itemize}\setlength\itemsep{0.65em}
    \item \textbf{Extralegal influences on judicial behavior} — Unlike prior U.S.-focused work linking sentencing to fiscal or electoral incentives {\footnotesize\color{gray}\citep{Ouss-Incentives, Dippel, Gordon-Judges-Election}}
    \begin{itemize}
        \item \alert{Mexican judges had neither} — yet still responded to prison capacity changes, making the findings especially compelling.
    \end{itemize}
\pause
    \item \textbf{Institutional constraints \& criminal justice} — Where U.S. evidence shows caseload pressure leads to fewer incarcerations and more plea deals due to time constraints {\footnotesize\color{gray}\citep{Yang-Resource-AEJ}}
    \begin{itemize}
        \item \alert{I find the opposite}: judges send more people to prison under exogenous caseload increases.
    \end{itemize}
\pause
    \item \textbf{Drivers of rising incarceration} — Adds to a largely U.S.-focused literature {\footnotesize\color{gray}\citep{Mukherjee-Private, Pfaff2007prison, Galinato-2020}} by identifying supply-side prison capacity as a causal driver of sentencing in a non-U.S. context.
\end{itemize}
\end{frame}

\section{Data}

\begin{frame}
\frametitle{Data}
\small
\begin{itemize}\setlength\itemsep{0.65em}
    \item Two administrative datasets covering all formally processed individuals in Mexico (local level):
    \begin{itemize}\setlength\itemsep{0.65em}
        \item \textbf{Pre-trial dataset}: early judicial decisions (detention vs.\ release)
        \item \textbf{Criminal courts dataset}: final outcomes (sentences, fines, acquittals)
    \end{itemize}
    \pause
    \item Both include rich individual-level information that lets me observe the \alert{municipality of residence}.
    \item Records aggregated to \alert{municipality $\times$ bimonth} level, matched with geolocated data on \alert{local prison capacity and population}.
    \pause
    \item \textbf{Interpretation (Generalized DiD):} The estimate compares how sentencing behavior changed in municipalities close to where a new prison opened — before versus after the opening — against municipalities where no prison opened over the same period.
\end{itemize}
\end{frame}
