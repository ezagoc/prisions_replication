\begin{figure}[!htpb]
   \centering
   \caption{Robustness to the log transformation (all outcomes in levels)}
   \label{fig:att_sent_robustness_chen}
   \includegraphics[width=1.1\textwidth]{figures/atts/robustness_chen_roth.pdf}
   \caption*{\footnotesize{\textit{Note:} Average Treatment on the Treated estimates from the effect of public prison construction on a 300KM vicinity on several outcomes measured at the date of sentencing. Four transformations of each outcome are applied: Inverse hyperbolic sine (IHS) transformation - $\sinh^{-1}(y) = \log(y + \sqrt{1 + y^2})$ (arcsinh), logarithmic (+1) transformation - $\log(y + 1)$ (log(y + 1)). binary indicator 1 if $y > 0$ and 0 otherwise ($1\{y>0\}$ and \cite{ChenRoth2024} transformation - $-\min\{y>0\}$ for $y=0$ and $\log(y)$ for $y>0$. Results are estimated using  \cite{Callaway} method with controls. 90\% (big) and 95\% (small) confidence intervals are plotted. Standard errors are clustered at the municipality level across all specifications. }}
\end{figure}
